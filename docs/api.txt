public class LessonDto
{
    public required string LessonId { get; set; }
    public required DateTime StartTime { get; set; }
    public required DateTime EndTime { get; set; }
    public required string Address { get; set; } // todo: replace with DTO

}

public class AttendanceDto
{
    public required string StudentName { get; set; }
    public required string StudentSurname { get; set; }
    public required bool? Confirmed { get; set; }
}

public class StudentDto
{
    public string? ExternalId { get; set; }
    public string? Name { get; set; }
    public string? Surname { get; set; }
    public AddressDto? Address { get; set; }
}

public class AddressDto
{
    public string? ExternalId { get; set; }
    public string? AddressName { get; set; }
    public string? AddressData { get; set; }
}

Planowanie lekcji:
POST http://localhost:3000/api/plan-lessons
posiadający body:
{
    "firstStartTime": "2025-06-02T12:00:00.0000000Z",
    "firstEndTime": "2025-06-02T14:00:00.0000000Z",
    "scheduleEndTime": "2025-07-28T10:00:00.0000000Z",
    "periodInDays": 7,
    "addressId": "101",
    "studentIds": ["201"]
}

private class PlanLessonsRequestBody
{
    public required DateTime FirstStartTime { get; set; }

    public required DateTime ScheduleEndTime { get; set; }
    public required int PeriodInDays { get; set; }
    public required string AddressId { get; set; }
    public required IList<string> StudentIds { get; set; }
}

STUDENT
Pobieranie ucznia po ID:
GET /api/student/{studentExternalId}
Zwraca: StudentDTO

Pobieranie uczniów:
GET /api/student - dla pobrania wszystkich uczniów
GET /api/student?lessonId=1 - dla pobrania uczniów powiązanych z lekcją o ID 1
Zwraca: Listę StudentDto

Tworzenie ucznia:
POST /api/student
Możliwe schematy Request Body :
{
    "Name": "Basia",
    "Surname": "Be",
    "Address": {
        "ExternalId": "2"
    }
}
Powyższe utworzy ucznia przypisując do niego istniejący już adres o ID 2
{
    "Name": "x",
    "Surname": "y",
    "Address": {
        "AddressName": "nowy",
        "AddressData": "ulica"
    }
}
Powyższe utworzy nowego ucznia oraz utworzy nowy adres
Zwraca: kod sukcesu/błędu

Update ucznia:
PATCH /api/student/{studentExternalId}
Możliwe schematy request body:
{
    "Name": "x",
    "Surname": "y",
    "Address": {
        "AddressName": "nowy",
        "AddressData": "ulica"
    }
}
Body może zawierać dowolne informacje spośród powyższych, pominięte dane nie zostaną zmienione.
UWAGA: Przy podaniu adresu jak poniżej:
{
    "Address": {
        "ExternalId": "2",
        "AddressName": "ZmianaNazwyIstniejącegoAdresu"
    }
}
Zostanie zmodyfikowany adres o ID 2
Natomiast przy podaniu adresu w poniższej formie:
{
    "Address": {
        "AddressName": "NazwaNowegoAdresu",
        "AddressData": "DaneNowegoAdresu"
    }
}
Zostanie utworzony nowy adres.
W sytuacji poniższej:
{
    "Address": {
        "AddressName": "NazwaNowegoAdresu"
    }
}
Zostanie utworzony nowy adres, a "AddressData" zostanie skopiowane z dotychczasowego adresu ucznia
Zwraca: kod sukcesu/błędu
